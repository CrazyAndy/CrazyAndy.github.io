列表视图：ListView.builder

import 'package:flutter/material.dart';
import 'model/post.dart';

void main() =>runApp(App());


class App extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Home(),
      theme: ThemeData(//主题
        primarySwatch: Colors.yellow //主要颜色
      ),
    );
  }
}

class Home extends StatelessWidget {

  Widget _listItemBuilder(BuildContext context,int index){
    //index是列表项目索引值
    return Text(
      posts[index].title
    );
  }

  @override
    Widget build(BuildContext context) {
      return Scaffold(
        appBar: AppBar(
          title: Text('CrazyAndy'),
          elevation: 30.0,
        ),
        body: ListView.builder(
          itemCount: posts.length,
          itemBuilder: _listItemBuilder,
        ),
      );
    }
}

列表项目

import 'package:flutter/material.dart';
import 'model/post.dart';

void main() =>runApp(App());


class App extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Home(),
      theme: ThemeData(//主题
        primarySwatch: Colors.yellow //主要颜色
      ),
    );
  }
}

class Home extends StatelessWidget {

  Widget _listItemBuilder(BuildContext context,int index){
    //index是列表项目索引值
    return Container(
      color: Colors.white,
      margin: EdgeInsets.all(8.0),//container的四周设置外边距
      child: Column(//竖着排列用column，横着用 row
        children: <Widget>[
          Image.network(posts[index].imageUrl),
          SizedBox(height: 16.0,),
          Text(
            posts[index].title,
            style:Theme.of(context).textTheme.title,//启用App中关于text的title的主题样式
          ),
          Text(
            posts[index].title,
            style:Theme.of(context).textTheme.subhead,//启用App中关于text的主题
          ),
          SizedBox(height: 16.0,),
        ],
      ),
    );
  }

  @override
    Widget build(BuildContext context) {
      return Scaffold(
        backgroundColor: Colors.grey[100],
        appBar: AppBar(
          title: Text('CrazyAndy'),
          elevation: 30.0,
        ),
        body: ListView.builder(
          itemCount: posts.length,
          itemBuilder: _listItemBuilder,
        ),
      );
    }
}
