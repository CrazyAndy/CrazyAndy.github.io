class App extends StatelessWidget {

Theme：定制、使用、重置、覆盖主题

在子页面中，默认使用父程序中的主题，但是我们可以自定义覆盖，重置主题：

import 'package:flutter/material.dart';

class FormDemo extends StatelessWidget {
  
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Theme(
        // data: ThemeData(//这里可以完全替代之前的主题
        //   primaryColor: Colors.black
        // ),
        data: Theme.of(context).copyWith(//这里可以覆盖某些特定的设置
          primaryColor: Colors.black//这样下来，除了主题颜色之外，其他的都会保留
        ),
        child: ThemeDemo(),
      ),
    );
  }
}

class ThemeDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      // color: Theme.of(context).primaryColor,
      color: Theme.of(context).accentColor,
    );
  }
}

Form：保存与获取表单里的数据

我们可以先给表单添加一个GlobalKey，然后把key交给表单，最后我们可以使用这个key来引用表单， 以此来获取表单里面输入的内容。这里的小部件我们要用 🈶️状态的小部件（以后会讲）StatefuleWidget,要不然程序会崩溃。

class RegisterForm extends StatefulWidget {

  _RegisterFormState createState() => _RegisterFormState();
}

class _RegisterFormState extends State<RegisterForm> {

  final registerFormKey =GlobalKey<FormState>();
  String username,password;

  void submitRegisterForm(){
    registerFormKey.currentState.save();
    debugPrint('username:$username');
    debugPrint('password:$password');
  }

  @override
  Widget build(BuildContext context) {
    return Form(
      key: registerFormKey,
      child: Column(
        children: <Widget>[
          TextFormField(
            //相当于在FormField里面包装了一个TextField

            onSaved: (value){
              username = value;
            },
          ),
          TextFormField(
            obscureText: true,
            decoration: InputDecoration(labelText: 'passowrd'),
            onSaved: (value){
              password = value;

          ),
          SizedBox(
            height: 32.0,
          ),

            width: double.infinity,
            child: RaisedButton(
              color: Theme.of(context).accentColor,

                'Register',
                style: TextStyle(color: Colors.white),
              ),
              elevation: 0.0,
              onPressed: submitRegisterForm,

          ),
        ],
      ),
    );
  }
}

Form：自动验证

自动验证功能也和上面风格类似，我们可以只允许其验证失败的时候再打开验证功能。这样下来就能防止刚进入登录页面的时候就验证。

class RegisterForm extends StatefulWidget {

  _RegisterFormState createState() => _RegisterFormState();
}

class _RegisterFormState extends State<RegisterForm> {

  final registerFormKey =GlobalKey<FormState>();
  String username,password;
  bool autovalidate =false;//自动验证功能默认关闭

  void submitRegisterForm(){
    if(registerFormKey.currentState.validate()){
      registerFormKey.currentState.save();
      debugPrint(' username:$username');
      debugPrint('password:$password');
    }else{
      autovalidate =true;
    }

    
  }

  String validateUsername(value){
    if(value.isEmpty){
      return 'username is required.';
    }
    return null;//表示验证成功
  }
  String validatePassword(value){
    if(value.isEmpty){
      return 'password is required';
    }
    return null;
  }

  @override
  Widget build(BuildContext context) {
    return Form(
      key: registerFormKey,
      child: Column(
        children: <Widget>[
          TextFormField(
            //相当于在FormField里面包装了一个TextField
            decoration: InputDecoration(
              labelText: 'UserName',
              helperText: ''),
            onSaved: (value){
              username = value;
            },
            validator: validateUsername,
            autovalidate: autovalidate,
          ),
          TextFormField(
            obscureText: true,
            decoration: InputDecoration(labelText: 'passowrd'),
            onSaved: (value){
              password = value;
            },
            validator: validatePassword,
            autovalidate: autovalidate,
          ),
          SizedBox(
            height: 32.0,
          ),
          Container(
            width: double.infinity,
            child: RaisedButton(
              color: Theme.of(context).accentColor,
              child: Text(
                'Register',
                style: TextStyle(color: Colors.white),
              ),
              elevation: 0.0,
              onPressed: submitRegisterForm,
            ),
          ),
        ],
      ),
    );
  }
}
