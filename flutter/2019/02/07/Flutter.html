BoxDecoration：装饰盒子

我们装饰盒子 在 container中用 decoration属性

BoxDecoration：边框（border）

decoration: BoxDecoration(//装饰盒子
                        color: Color.fromRGBO(3, 54, 255, 1.0),//如果这里设置了背景颜色，就要把Container里面的color属性给注释掉
                        border: Border(//边框设置
                          top: BorderSide(
                            color: Colors.indigoAccent[100],
                            width: 3.0,
                            style: BorderStyle.solid
                          )
                        )
                      ),

BoxDecoration：圆角（borderRadius）

BoxDecoration：阴影（borderShadow）

borderRadius: BorderRadius.circular(26.0), //圆角效果
                          boxShadow: [
                            BoxShadow(
                              offset: Offset(
                                  6.0, 7.0), //控制阴影的偏移，第一个参数为x轴偏移量，第二个是y轴偏移量
                              color: Color.fromRGBO(
                                  16, 20, 188, 1.0), //默认阴影颜色是黑色，这里可以做修改
                              blurRadius: 1, //阴影的模糊程度，值越大，效果越明显
                              spreadRadius: 3.0, //正数是扩大阴影面积，负数是缩小
                            )
                          ]),

BoxDecoration：形状（shape）

BoxDecoration：渐变（gradient）

BoxDecoration：背景图像（image）

class ContainerDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
          image: DecorationImage(//这里设置背景图像
            image: NetworkImage('https://ss3.baidu.com/-fo3dSag_xI4khGko9WTAnF6hhy/image/h%3D300/sign=e8ce788890510fb367197197e933c893/b999a9014c086e061396097a0c087bf40ad1cbc0.jpg'),
            alignment: Alignment.topCenter,
            fit: BoxFit.cover, //扩充
        )),
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: <Widget>[
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: <Widget>[
              Container(
                color: Colors.grey[100], //背景颜色
                child: Row(
                  //row 里面可以添加一组横排显示的部件
                  children: <Widget>[
                    Container(
                      child: Icon(Icons.pool, size: 32.0, color: Colors.white),
                      // color: Color.fromRGBO(3, 54, 255, 1.0), //背景颜色,最后一个是不透明度
                      // padding: EdgeInsets.only(left: 24,right: 8),
                      padding: EdgeInsets.all(16.0), //内边距
                      margin: EdgeInsets.all(8.0), //外边距
                      width: 90.0,
                      height: 90.0,
                      decoration: BoxDecoration(
                        //装饰盒子
                        color: Color.fromRGBO(3, 54, 255,
                            1.0), //如果这里设置了背景颜色，就要把Container里面的color属性给注释掉
                        border: Border(//边框设置
                            // top: BorderSide(
                            //   color: Colors.indigoAccent[100],
                            //   width: 1.0,
                            //   style: BorderStyle.solid
                            // )
                            ),
                        // borderRadius: BorderRadius.circular(26.0), //圆角效果
                        boxShadow: [
                          BoxShadow(
                            offset: Offset(
                                6.0, 7.0), //控制阴影的偏移，第一个参数为x轴偏移量，第二个是y轴偏移量
                            color: Color.fromRGBO(
                                16, 20, 188, 1.0), //默认阴影颜色是黑色，这里可以做修改
                            blurRadius: 1, //阴影的模糊程度，值越大，效果越明显
                            spreadRadius: 3.0, //正数是扩大阴影面积，负数是缩小
                          )
                        ],
                        shape: BoxShape
                            .circle, //如果这里使用圆形 形状，那么就不能使用 圆角效果，需要将上面 borderRadius给注释掉
                        // gradient: RadialGradient(//渐变效果，有三种，自己探究
                        //     colors: [
                        //   Color.fromRGBO(7, 102, 255, 1.0),
                        //   Color.fromRGBO(3, 28, 128, 1.0),
                        // ])
                      ),
                    )
                  ],
                ),
              )
            ],
          )
        ],
      ),
    );
  }
}
