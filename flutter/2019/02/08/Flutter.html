Flutter基础学习——布局004

准备

创建LayoutDemo小部件

创建可配置的图标徽章(IconBadge)小部件

import 'package:flutter/material.dart';


class LayoutDemo extends StatelessWidget {

  @override
  Widget build(BuildContext context) {
    return Column(
      children: <Widget>[
        IconBadge(Icons.pool,size: 64,),
      ],
    );
  }
  
}

class IconBadge extends StatelessWidget {

  final IconData icon;
  final double size;

  IconBadge(this.icon,{this.size=32.0});
  
  @override
  Widget build(BuildContext context) {
    return Container(
      child: Icon(icon,size:size,color:Colors.white),
      width: size + 60,
      height: size + 60,
      color: Color.fromRGBO(3, 54, 255, 1.0),
    );
  }

}

布局

Row(横排)与Column(竖排)

row为横着排，column 是竖着排，借用bootstrap css样式中 flexable 布局概念。

mainAxis：主轴

Row 的主轴是横着的，Column主轴是竖着的。

mainAxisAlignment: MainAxisAlignment.start

return Container(
      child: Column(
        // mainAxisAlignment: MainAxisAlignment.start,//从主轴开始往后排
        // mainAxisAlignment: MainAxisAlignment.end,//从主轴结束往前排
        // mainAxisAlignment: MainAxisAlignment.center,//主轴中间
        // mainAxisAlignment: MainAxisAlignment.spaceAround,//将主轴剩下空间分配到小部件的周围
        // mainAxisAlignment: MainAxisAlignment.spaceBetween,//将主轴剩下空间分配到小部件之间
        mainAxisAlignment: MainAxisAlignment.spaceEvenly, //将主轴剩下的空间平均分配到小部件之间
        children: <Widget>[
          IconBadge(
            Icons.pool,
            size: 64,
          ),
          IconBadge(
            Icons.beach_access,
            size: 64,
          ),
          IconBadge(
            Icons.airplanemode_active,
            size: 64,
          ),
        ],
      ),
    );

crossAxis：交叉轴

crossAxisAlignment: CrossAxisAlignment.start

交叉轴即为主轴的垂直轴，比方说 Row的小部件中，主轴是 横轴，那么交叉轴就是竖zhou

SizedBox：固定尺寸的盒子

SizeBox我们可以用来包裹 widget，然后用sizebox的宽高来定义区域块的宽高，相当于在widget外的container 定义 padding；当然可以使用一个只有高度或宽度的 SizeBox来隔开两个小部件。

import 'package:flutter/material.dart';

class LayoutDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: <Widget>[
          SizedBox(
            width: 200.0,
            height: 300.0,
            child: Container(
              decoration: BoxDecoration(
                color: Color.fromRGBO(3, 54, 255, 1.0),
                borderRadius: BorderRadius.circular(8.0),
              ),
              child: Icon(
                Icons.ac_unit,color:Colors.white,size:32.0
              ),
            ),
          ),
          SizedBox(height: 32.0,),
          SizedBox(
            width: 100.0,
            height: 100.0,
            child: Container(
              decoration: BoxDecoration(
                color: Color.fromRGBO(3, 54, 255, 1.0),
                borderRadius: BorderRadius.circular(8.0),
              ),
              child: Icon(
                Icons.brightness_2,color:Colors.white,size:32.0
              ),
            ),
          ),
        ],
      ),
    );
  }
}

Alignment：对齐

Container 有个属性是 alignment: Alignment(0, 1.0),我们可以根据这个设置子部件在container中的百分比位置，

Alignment（x,y） 两个参数代表 横竖方向的比例，如果-1<x<0,那么部件应该在横向的左半部某个位置。

Stack：一摞小部件

statck会将子部件都叠加到一块，也可在stack的子部件添加 Positioned小部件(包含left, right,top,bottom属性,相对于父部件内部位置)

AspectRatio：宽高比

设置子部件宽高 比例

ConstrainedBox：带限制的盒子

ConstrainedBox 可是设置最小 最大 宽高

本章所有代码整理如下：

import 'package:flutter/material.dart';

class LayoutDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: <Widget>[
          Stack(
            alignment: Alignment.topLeft,
            children: <Widget>[
              SizedBox(
                width: 200.0,
                height: 300.0,
                child: Container(
                  alignment: Alignment(0, 1.0),
                  decoration: BoxDecoration(
                    color: Color.fromRGBO(3, 54, 255, 1.0),
                    borderRadius: BorderRadius.circular(8.0),
                  ),
                  child: Icon(Icons.ac_unit, color: Colors.white, size: 32.0),
                ),
              ),
              SizedBox(
                height: 32.0,
              ),
              SizedBox(
                width: 100.0,
                height: 100.0,
                child: Container(
                  decoration: BoxDecoration(
                      color: Color.fromRGBO(3, 54, 255, 1.0),
                      borderRadius: BorderRadius.circular(8.0),
                      gradient: RadialGradient(colors: [
                        Color.fromRGBO(7, 102, 255, 1.0),
                        Color.fromRGBO(3, 54, 255, 1.0),
                      ])),
                  child:
                      Icon(Icons.brightness_2, color: Colors.white, size: 32.0),
                ),
              ),
              Positioned(
                right: 20.0,
                child: Icon(Icons.ac_unit, color: Colors.white, size: 32.0),
              )
            ],
          ),
        ],
      ),
    );
  }
}

class AspectRatioDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: <Widget>[
          AspectRatio(
            aspectRatio: 16/9,
            child: Container(
              color: Color.fromRGBO(3, 54, 255, 1.0),
            ),
          )
        ],
      ),
    );
  }
}

class ConstraintDemo extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: <Widget>[
          ConstrainedBox(
            constraints: BoxConstraints(
              minHeight: 200.0,
            ),
            child: Container(
              color: Color.fromRGBO(3, 54, 255, 1.0),
            ),
          )
        ],
      ),
    );
  }
}
